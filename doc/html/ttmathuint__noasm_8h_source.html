588  template<uint value_size>
589  void UInt<value_size>::MulTwoWords(uint a, uint b, uint * result_high, uint * result_low)
590  {
591 #ifdef TTMATH_PLATFORM32
592 
593  /*
594  on 32bit platforms we have defined 'unsigned long long int' type known as 'ulint' in ttmath namespace
595  this type has 64 bits, then we're using only one multiplication: 32bit * 32bit = 64bit
596  */
597 
598  union uint_
599  {
600  struct
601  {
602  uint low; // 32 bits
603  uint high; // 32 bits
604  }u_;
605 
606  ulint u; // 64 bits
607  }res;
608 
609  res.u = ulint(a) * ulint(b); // multiply two 32bit words, the result has 64 bits
610 
611  *result_high = res.u_.high;
612  *result_low = res.u_.low;
613 
614 #else
615 
616  /*
617  64 bits platforms
618 
619  we don't have a native type which has 128 bits
620  then we're splitting 'a' and 'b' to 4 parts (high and low halves)
621  and using 4 multiplications (with additions and carry correctness)
622  */
623 
624  uint_ a_;
625  uint_ b_;
626  uint_ res_high1, res_high2;
627  uint_ res_low1, res_low2;
628 
629  a_.u = a;
630  b_.u = b;
631 
632  /*
633  the multiplication is as follows (schoolbook algorithm with O(n^2) ):
634 
635  32 bits 32 bits
636 
637  +--------------------------------+
638  | a_.u_.high | a_.u_.low |
639  +--------------------------------+
640  | b_.u_.high | b_.u_.low |
641  +--------------------------------+--------------------------------+
642  | res_high1.u | res_low1.u |
643  +--------------------------------+--------------------------------+
644  | res_high2.u | res_low2.u |
645  +--------------------------------+--------------------------------+
646 
647  64 bits 64 bits
648  */
649 
650  uint_ temp;
651 
652  res_low1.u = uint(b_.u_.low) * uint(a_.u_.low);
653 
654  temp.u = uint(res_low1.u_.high) + uint(b_.u_.low) * uint(a_.u_.high);
655  res_low1.u_.high = temp.u_.low;
656  res_high1.u_.low = temp.u_.high;
657  res_high1.u_.high = 0;
658 
659  res_low2.u_.low = 0;
660  temp.u = uint(b_.u_.high) * uint(a_.u_.low);
661  res_low2.u_.high = temp.u_.low;
662 
663  res_high2.u = uint(b_.u_.high) * uint(a_.u_.high) + uint(temp.u_.high);
664 
665  uint c = AddTwoWords(res_low1.u, res_low2.u, 0, &res_low2.u);
666  AddTwoWords(res_high1.u, res_high2.u, c, &res_high2.u); // there is no carry from here
667 
668  *result_high = res_high2.u;
669  *result_low = res_low2.u;
670 
671 #endif
672  }

693  void UInt<value_size>::DivTwoWords(uint a, uint b, uint c, uint * r, uint * rest)
694  {
695  // (a < c ) for the result to be one word
696  TTMATH_ASSERT( c != 0 && a < c )
697 
698 #ifdef TTMATH_PLATFORM32
699 
700  union

705  uint high; // 32 bits

708  ulint u; // 64 bits
709  }ab;
710 
711  ab.u_.high = a;
712  ab.u_.low = b;

714  *r = uint(ab.u / c);
715  *rest = uint(ab.u % c);
716 
717 #else
718 
719  uint_ c_;
720  c_.u = c;
721 
722  if( a == 0 )
723  {
724  *r = b / c;
725  *rest = b % c;
726  }
727  else
728  if( c_.u_.high == 0 )
729  {

738  temp1.u_.low = b_.u_.high;

741  temp2.u_.high = (unsigned int)(temp1.u % c);
742  temp2.u_.low = b_.u_.low;

745  *rest = temp2.u % c;

748  }

765  template<uint value_size>
766  void UInt<value_size>::DivTwoWords2(uint a, uint b, uint c, uint * r, uint * rest)
767  {
768  // a is not zero
769  // c_.u_.high is not zero
770 
771  uint_ a_, b_, c_, u_, q_;
772  unsigned int u3; // 32 bit
773 
774  a_.u = a;
775  b_.u = b;
776  c_.u = c;
777 
778  // normalizing
779  uint d = DivTwoWordsNormalize(a_, b_, c_);
780 
781  // loop from j=1 to j=0
782  // the first step (for j=2) is skipped because our result is only in one word,
783  // (first 'q' were 0 and nothing would be changed)
784  u_.u_.high = a_.u_.high;
785  u_.u_.low = a_.u_.low;
786  u3 = b_.u_.high;
787  q_.u_.high = DivTwoWordsCalculate(u_, u3, c_);
788  MultiplySubtract(u_, u3, q_.u_.high, c_);
789 
790  u_.u_.high = u_.u_.low;
791  u_.u_.low = u3;
792  u3 = b_.u_.low;
793  q_.u_.low = DivTwoWordsCalculate(u_, u3, c_);
794  MultiplySubtract(u_, u3, q_.u_.low, c_);
795 
796  *r = q_.u;
797 
798  // unnormalizing for the remainder
799  u_.u_.high = u_.u_.low;
800  u_.u_.low = u3;
801  *rest = DivTwoWordsUnnormalize(u_.u, d);
802  }

804  template<uint value_size>
805  uint UInt<value_size>::DivTwoWordsNormalize(uint_ & a_, uint_ & b_, uint_ & c_)
806  {
807  uint d = 0;
808 
809  for(; (c_.u & TTMATH_UINT_HIGHEST_BIT) == 0; ++d )
810  {
811  c_.u = c_.u << 1;
812 
813  uint bc = b_.u & TTMATH_UINT_HIGHEST_BIT; // carry from 'b'

816  a_.u = a_.u << 1; // carry bits from 'a' are simply skipped
817 
818  if( bc )
819  a_.u = a_.u | 1;
820  }
821 
822  return d;

836  template<uint value_size>
837  unsigned int UInt<value_size>::DivTwoWordsCalculate(uint_ u_, unsigned int u3, uint_ v_)
838  {
839  bool next_test;
840  uint_ qp_, rp_, temp_;
841 
842  qp_.u = u_.u / uint(v_.u_.high);
843  rp_.u = u_.u % uint(v_.u_.high);
844 
845  TTMATH_ASSERT( qp_.u_.high==0 || qp_.u_.high==1 )
846 
847  do
848  {
849  bool decrease = false;
850 
851  if( qp_.u_.high == 1 )
852  decrease = true;
853  else
854  {
855  temp_.u_.high = rp_.u_.low;
856  temp_.u_.low = u3;
857 
858  if( qp_.u * uint(v_.u_.low) > temp_.u )
859  decrease = true;
860  }
861 
862  next_test = false;
863 
864  if( decrease )
865  {
866  --qp_.u;
867  rp_.u += v_.u_.high;
868 
869  if( rp_.u_.high == 0 )
870  next_test = true;
871  }
872  }
873  while( next_test );
874 
875  return qp_.u_.low;
