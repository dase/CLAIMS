40 template<class Archive, class Container, class InputFunction>
41 inline void load_unordered_collection(Archive & ar, Container &s)
42 {
43  s.clear();
44  collection_size_type count;
45  collection_size_type bucket_count;
46  item_version_type item_version(0);
47  boost::archive::library_version_type library_version(
48  ar.get_library_version()
49  );
50  // retrieve number of elements
51  ar >> BOOST_SERIALIZATION_NVP(count);
52  ar >> BOOST_SERIALIZATION_NVP(bucket_count);
53  if(boost::archive::library_version_type(3) < library_version){
54  ar >> BOOST_SERIALIZATION_NVP(item_version);
55  }
56  s.rehash(bucket_count);
57  InputFunction ifunc;
58  while(count-- > 0){
59  ifunc(ar, s, item_version);
60  }
61 }

63 template<class Archive, class Container>
64 inline void save_unordered_collection(Archive & ar, const Container &s)
65 {
66  collection_size_type count(s.size());
67  const collection_size_type bucket_count(s.bucket_count());
68  const item_version_type item_version(
69  version<BOOST_DEDUCED_TYPENAME Container::value_type>::value
70  );
71 
72  #if 0
73  /* should only be necessary to create archives of previous versions
74  * which is not currently supported. So for now comment this out
75  */
76  boost::archive::library_version_type library_version(
77  ar.get_library_version()
78  );
79  // retrieve number of elements
80  ar << BOOST_SERIALIZATION_NVP(count);
81  ar << BOOST_SERIALIZATION_NVP(bucket_count);
82  if(boost::archive::library_version_type(3) < library_version){
83  // record number of elements
84  // make sure the target type is registered so we can retrieve
85  // the version when we load
86  ar << BOOST_SERIALIZATION_NVP(item_version);
87  }
88  #else
89  ar << BOOST_SERIALIZATION_NVP(count);
90  ar << BOOST_SERIALIZATION_NVP(bucket_count);
91  ar << BOOST_SERIALIZATION_NVP(item_version);
92  #endif
93 
94  BOOST_DEDUCED_TYPENAME Container::const_iterator it = s.begin();
95  while(count-- > 0){
96  // note borland emits a no-op without the explicit namespace
97  boost::serialization::save_construct_data_adl(
98  ar,
99  &(*it),
100  boost::serialization::version<
101  BOOST_DEDUCED_TYPENAME Container::value_type
102  >::value
103  );
104  ar << boost::serialization::make_nvp("item", *it++);
105  }
106 }

110 struct archive_input_unordered_map
111 {
112  inline void operator()(
113  Archive &ar,
114  Container &s,
115  const unsigned int v
116  ){
117  typedef BOOST_DEDUCED_TYPENAME Container::value_type type;
118  detail::stack_construct<Archive, type> t(ar, v);
119  // borland fails silently w/o full namespace
120  ar >> boost::serialization::make_nvp("item", t.reference());
121  std::pair<BOOST_DEDUCED_TYPENAME Container::const_iterator, bool> result =
122  s.insert(t.reference());
123  // note: the following presumes that the map::value_type was NOT tracked
124  // in the archive. This is the usual case, but here there is no way
125  // to determine that.
126  if(result.second){
127  ar.reset_object_address(
128  & (result.first->second),
129  & t.reference().second
130  );
131  }
132  }
133 };
